Execute Remote Script File by PowerShell

Invoke-RestMethod -Uri "http://IP-Address:19312/executescript" -Method Post -Body (@{"ScriptLocation"=[Convert]::ToBase64String([System.Text.Encoding]::UTF8.GetBytes("Script-File-Name.xml"))}|ConvertTo-JSON)

or

$uri = "http://IP-Address:19312/executescript"
$file = "Script-File-Name.xml"
$b64file = [Convert]::ToBase64String([System.Text.Encoding]::UTF8.GetBytes($file))
Invoke-RestMethod -Uri $uri -Method Post -Body (@{"ScriptLocation"=$b64file}|ConvertTo-JSON)
